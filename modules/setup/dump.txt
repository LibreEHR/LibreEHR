open browser via terminal script

======
- mac
======
	$ open -a /Applications/Safari.app http://www.google.com
	$ open -a /Applications/Firefox.app http://www.google.com
	$ open some_url
	
======
-linux
======

#!/bin/bash
if which xdg-open > /dev/null
then
  xdg-open URL
elif which gnome-open > /dev/null
then
  gnome-open URL
fi

// command to open browser and exit  terminal
nohup xdg-open http://www.google.com  &  exit

//open browser with parameters
xdg-open http://localhost/test.php?step=46
